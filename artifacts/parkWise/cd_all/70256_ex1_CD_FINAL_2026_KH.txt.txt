class ParkingLot {
    +int lotNumber
    +string name
    +string address
    +string city
    +numAvaSpaces
    +int availableSpaces

    +generateAnnualSummary(year : int)
}

class ParkingSpace {
    +spaceId
    +floor
    +size : ParkingSpaceSize
    +pos x
    +pos y
    +isVacant
}

enum ParkingSpaceSize {
    small
    medium
    large
}

class Conveyor {
    +conveyorId
    +floor
    +pos x
    +pos y
    +maxWeight

}

    +priceListId
    +firstHourPrice
    +additionalHourPrice
    +fullDayPrice

}

class Client {
    +clientId
    +firstName
    +lastName
    +phoneNumber
}

class Membership {
    +membershipNum
    +membershipStartDate
    +discountPercent
    +isActive
}

class PreferredLotSelection {
    +selectionNum
    +selectionDate
}

class Vehicle {
    +vehicleNumber
    +type
    +weight
    +vehSize : VehicleSize
    +vehColor : VehicleColor
    +photoFront
    +photoRear
    +photoRightSide
    +photoLeftSide
}

enum VehicleSize {
    small
    medium
    large
}

enum VehicleColor {
    black
    white
    silver
    blue
    other
}

class ParkingSession {
    +sessionId
    +date
    +startTime
    +endTime
    +totalHours
    +finalAmount
    +isFree
}


class Payment {
    +paymentId
    +amount
    +paymentTime

}
