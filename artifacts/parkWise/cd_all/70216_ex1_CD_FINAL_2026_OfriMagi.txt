}

}

  endTime

}

  address

  availableSpaces
}

  id

  positionX
  positionY

}

' ===================== RELATIONSHIPS =====================

' Customer ↔ Membership
Customer "1" -- "0..1" Membership

' Customer ↔ Vehicle
Customer "1" -- "0..*" Vehicle
Vehicle "1" -- "1" Customer

' Membership ↔ PreferredParkingLot
Membership "1" -- "0..*" PreferredParkingLot
PreferredParkingLot "1" -- "1" ParkingLot

' Vehicle ↔ VehiclePhoto
Vehicle "1" *-- "0..3" VehiclePhoto

' Vehicle ↔ ParkingSession
Vehicle "1" -- "0..*" ParkingSession
ParkingSession "1" -- "1" Vehicle

' ParkingSession relations
ParkingSession "1" -- "1" ParkingSpot
ParkingSession "1" -- "1" ParkingLot
ParkingSession "1" -- "0..1" Conveyor
ParkingSession "1" -- "1" PriceList

' ParkingLot compositions
ParkingLot "1" -- "0.." ParkingSpot
ParkingLot "1" -- "0.." Conveyor

' Price history
ParkingLot "1" -- "0..*" PriceHistory
PriceList "1" o-- "0..*" PriceHistory

' External JSON service
RateService ..> PriceList : "provides JSON"

Receipt "1" -- "1" Customer
