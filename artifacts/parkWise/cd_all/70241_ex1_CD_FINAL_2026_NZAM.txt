' ENUMS
' =======================
enum VehicleSize {
  SMALL
  MEDIUM
  LARGE
}

enum VehicleColor {
  WHITE
  BLACK
  SILVER
  BLUE
  OTHER
}

enum SessionStatus {
  ACTIVE
  COMPLETED
  CANCELLED
}

enum PaymentStatus {
  PENDING
  APPROVED
  REJECTED
}

' =======================
' CORE DOMAIN ENTITIES
' =======================
class ParkingLot {
  - lotId: int
  - lotName: String
  - lotAddress: String
  - lotCity: String
  - availableSpaces: int
}

class ParkingSpace {
  - spaceId: int
  - spaceFloor: int
  - x: int
  - y: int
  - size: VehicleSize
  - isOccupied: boolean
}

class Conveyor {
  - conveyorId: int
  - maxWeightKg: double
  - currentFloor: int
  - currentX: int
  - currentY: int
  - isAvailable: boolean
}

class GateSensor {
  - sensorId: int
}

class Vehicle {
  - vehicleId: int
  - plateNumber: String
  - vehicleType: String
  - vehicleColor: VehicleColor
  - vehicleSize: VehicleSize
  - weightKg: double
}

class ParkingSession {
  - sessionId: int
  - startTime: DateTime
  - endTime: DateTime
  - totalHours: double
  - pSessionStatus: SessionStatus
}

class PriceList {
  - priceListId: int
  - description: String
  - activeFrom: Date
  - activeTo: Date
  - firstHourPrice: double
  - additionalHourPrice: double
  - fullDayPrice: double
}

class Payment {
  - paymentId: int
  - paymentAmount: double
  - paymentDate: DateTime
  - paymentStatus: PaymentStatus
}

' =======================
' RECEIPT
' =======================
class Receipt {
  - receiptId: int
  - issueDate: DateTime
  - totalHours: double
  - finalAmount: double
}

' =======================
' CLIENT CLUB
' =======================
class ClubMembership {
  - membershipId: int
  - registrationDate: Date
  - discountRate: double
  - personId: int
}

class PreferredParkingLot {
  - preferredId: int
  - lotId: int
  - selectionDate: Date
  - membershipId: int
}

' =======================
' REPORTING ENTITIES
' =======================
class AnnualSummaryReport {
  - reportForAllId: int
  - reportYear: int
  - generatedAt: DateTime
  +addLot(line: LotReport): void
}

class LotReport {
  - lotReportId: int
  - totalPaid: double
  - otherColorVehicles: int
  - lotId: int
}

' =======================
' SYSTEM
' =======================
class System {
  +getAllParkingLots(): ParkingLot[]
  +getAllSessions(): ParkingSession[]
  +importParkingData(): void
  +generateAnnualSummaryReport(year: int): AnnualSummaryReport
  +startParkingSession(vehicle: Vehicle, lot: ParkingLot, space: ParkingSpace): ParkingSession
}
