  +vehicleId : String
  +licenseNumber : String
  +type : String

  +size : SizeType
  +weight : int
}

  white
  black
  silver
  blue
  other
}

  +photoId : String
  +url : String

}

enum ViewType {
  front
  rear
  side
}

  +sessionId : String

  +calculatePrice()
