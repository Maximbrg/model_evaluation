    - companyName: String

    + registerClient(c: Client)
    + addParkingLot(pl: ParkingLot)
    + calculateCost(session: ParkingSession): double

    - clientID: int
    - firstName: String
    - lastName: String
    - phoneNumber: String
    + requestMembership()
    + addVehicle(v: Vehicle)
    + endParking()
    + pay()

    - membershipId: String

    + isEligibleForFreeParking(date: Date): boolean
}

class Vehicle {
    - vehicleNumber: String
    - clientID: String
    - type: String
    - size: Size
    - color: Color
    - weight: double
    - photos: List<image>
}

class ParkingLot {
    - uniqueNumber: String
    - name: String
    - street: String
    - streetNumber: int
    - city: String
    - availableSpaces: int

    + getCurrentPriceList(): PriceList
}

class PriceList {
    - effectiveDate: Date
    - firstHourPrice: double
    - additionalHourPrice: double
    - fullDayPrice: double
    + parseFromJSON(json: String)
}

    - exitTime: DateTime

    - appliedRate: String

}

class ParkingSpace {

    - size: Size
    - floorNumber: int
    - x: int
    - y: int

}

class Conveyor {
    - ConveyorID: int
    - floorNumber: int
    - x: int
    - y: int

}

' System & Admin

ParkWise "1" o-- "*" Client : manages >
Administrator ..> ParkWise : generates reports from >
Administrator ..> ParkingLot : updates details >

' Price List
ParkingLot "1" *-- "*" PriceList : maintains history >

' Membership
Client "1" *-- "0..1" ClubMembership : has >
ClubMembership "*" -- "*" ParkingLot
(ClubMembership, ParkingLot) .. PreferredSelection

' Parking Session
Vehicle "1" -- "*" ParkingLot
(Vehicle, ParkingLot) .. ParkingSession
ParkingSession "*" --> "1" Client : billed to

' Receipt
ParkingSession "1" *-- "0..1" Receipt : generates >

' Enum Connections
Vehicle ..> Color : has >
Vehicle ..> Size : has >
ParkingSpace ..> Size : restricted by >


' Composition & Dependencies
ParkingLot "1" *-- "*" ParkingSpace : contains >
ParkingSession "0..1" --> "1" ParkingSpace : occupies
ParkingSession "0..1" --> "1" Conveyor : transported by

ParkingLot ..> IGateSensor
