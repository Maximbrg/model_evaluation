' ---------------------------------------------------------
' 1. Class Definitions
' ---------------------------------------------------------

class "ParkWiseSystem" as System {
    + generateAnnualSummaryReport()
    + generateClubGrowthReport()
}

class "Administrator" as Administrator {
    - adminID
    - name
    + updateParkingLotDetails(lotID)
    + requestAnnualReport()
    + requestClubGrowthReport()
}

class "ParkingLot" as ParkingLot {
    - lotID
    - name
    - city
    - address
    - availableSpaces
    + calculateAnnualIncome()
    + countOtherColorVehicles()
}

class "Client" as Client {
    - clientID
    - firstName
    - lastName
    - phone
    - isClubMember
    - joinDate
    + registerToClub()
}

class "ParkingSpace" as ParkingSpace {
    - spaceID
    - floor
    - coordinates_XY
    - size: Size
}

class "Conveyor" as Conveyor {
    - conveyorID
    - maxWeight
}

class "GateSensor" as GateSensor {
    - sensorID
    + openGate()
}

class "PriceList" as PriceList {
    - listID
    - year
    - prices
}

class "Vehicle" as Vehicle {
    - licensePlate
    - type
    - weight
    - color: Color
    + extractDetailsFromPhoto()
}

class "ParkingSession" as ParkingSession {
    - sessionID
    - entryTime
    - exitTime
    - totalAmount
    + calculateFinalCost()
}

class "Size" as EnumSize <<enumeration>> {
    SMALL
    MEDIUM
    LARGE
}

class "Color" as EnumColor <<enumeration>> {
    WHITE
    BLACK
    SILVER
    BLUE
    OTHER
}
