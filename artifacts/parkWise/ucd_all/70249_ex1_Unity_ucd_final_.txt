' -------------------------------
' Actors
' -------------------------------
actor Client as C

actor IRS
actor "SMS Service" as SMS

' -------------------------------
' System Boundary
' -------------------------------
rectangle "ParkWise System" {

    ' ===== CLIENT FUNCTIONS =====
    rectangle "Client Functions" {

        usecase "Join Customer Club" as UC_JoinClub
        usecase "End Parking Session" as UC_End
        usecase "Access System\n(Phone Number)" as UC_Access
        usecase "View Parking Details" as UC_ViewParking
        usecase "Register Client" as UC_Register
        usecase "Upload Vehicle Photos" as UC_Photos
        usecase "Select Preferred Lots" as UC_SelectLots
        usecase "Pay for Parking" as UC_Pay

        usecase "Extract Vehicle Details\nFrom Photos" as UC_Extract
        usecase "Associate Vehicle\nWith Client" as UC_Associate
    }

    ' ===== GATE & ARRIVAL =====
    rectangle "Gate & Arrival Handling" {

        usecase "Verify Vehicle Number" as UC_VerifyVehicle
        usecase "Open Barrier" as UC_OpenGate
        usecase "Detect Arrival/Departure" as UC_Detect
    }

    ' ===== ADMIN FUNCTIONS =====
    rectangle "Admin Functions" {

        usecase "Upload Price List JSON" as UC_UploadPrice
        usecase "Update Parking Lot Details" as UC_UpdateLots
        usecase "Generate Annual Summary Report\n[PDF + XML]" as UC_AnnualReport
        usecase "Generate Club Growth Report" as UC_ClubGrowthReport
    }

    ' ===== CONVEYOR OPERATIONS =====
    rectangle "Conveyor Operations" {

        usecase "Retrieve Vehicle To Gate" as UC_Retrieve
        usecase "Assign Conveyor" as UC_Assign
        usecase "Move Vehicle To Space" as UC_Move
        usecase "Find Nearest Suitable\nParking Space" as UC_Optimize
        usecase "Scan Vehicle Number" as UC_Scan

        ' new improvement UCs
        usecase "Update Vehicle Location" as UC_UpdateLocation
        usecase "Send SMS Notification" as UC_SendSMS
    }

    ' ===== PAYMENT HANDLING =====
    rectangle "Payment Handling" {

        usecase "Issue Receipt" as UC_Receipt
        usecase "Process Payment Request" as UC_ProcessPayment
    }
}

' -------------------------------

' Verify → Open Barrier

' Assign conveyor → scan & optimize
UC_Assign --> UC_Optimize       : <<include>>
UC_Assign --> UC_Scan           : <<include>>

' Move vehicle → update location → send SMS
UC_Move --> UC_UpdateLocation   : <<include>>
UC_UpdateLocation --> UC_SendSMS : <<include>>

' End parking → retrieve vehicle & pay
UC_End --> UC_Retrieve          : <<include>>
UC_End --> UC_Pay               : <<include>>

' Payment → gateway & optional receipt
UC_Pay --> UC_ProcessPayment    : <<include>>
UC_Pay --> UC_Receipt           : <<extend>>
