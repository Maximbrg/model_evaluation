rectangle "ParkWise System" {

  ' --- Core parking flow ---
  usecase "Handle vehicle arrival" as UC_Entry

  usecase "Update available spaces" as UC_UpdateSpaces
  usecase "Manage parking session\n----------------------\nextension points:\n- free session for club member\n- conveyor problem" as UC_StartSession

  usecase "End parking & charge client\n----------------------\nextension points:\n- payment declined" as UC_EndSession
  usecase "Calculate parking fee" as UC_CalcFee
  usecase "Process payment" as UC_Payment
  usecase "Issue receipt" as UC_Receipt

  ' --- Client club / preferences ---
  usecase "Register to Client Club" as UC_RegisterClub
  usecase "Manage Preferred Parking Lots" as UC_ManagePreferred

  ' --- Admin / configuration / reports ---
  usecase "Update parking lot data" as UC_MaintainLots
  usecase "Receive yearly price lists (JSON) from IRS" as UC_ManagePrices
  usecase "Adopt price list for parking lots" as UC_AdoptPrices
  usecase "Generate annual summary report" as UC_AnnualReport
  usecase "Generate customer club growth report" as UC_ClubReport
}
