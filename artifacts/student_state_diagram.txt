@startuml

[*] --> OrderPlacement

state OrderPlacement {
    [*] --> RegularOrder
    RegularOrder --> NotifyGalGalim : Place regular order
    NotifyGalGalim --> CheckDeliveryConditions : Verify delivery conditions

    [*] --> UrgentOrder
    UrgentOrder --> DispatchDelivery : Initiate immediate delivery
}

CheckDeliveryConditions --> AddToDelivery : Fits delivery limits
CheckDeliveryConditions --> CreateNewDelivery : No suitable delivery exists
CheckDeliveryConditions --> RejectOrder : Below minimum or exceeds maximum limits

AddToDelivery --> ReadyForDispatch : Consolidated into existing delivery
CreateNewDelivery --> ReadyForDispatch : New delivery created
ReadyForDispatch --> DispatchDelivery : Dispatch triggered (bottles ≥ minimum OR explicit request)

state Transit {
    [*] --> InTransit
    InTransit --> IssueDetected : Delivery issue arises
    IssueDetected --> ResolveIssue : Urgent order
    IssueDetected --> CancelDelivery : Regular order
    ResolveIssue --> InTransit : Resume delivery
    CancelDelivery --> CreateNewDelivery : New delivery for remaining orders
    InTransit --> DeliveredSuccessfully : Delivery completed
}

DispatchDelivery --> Transit

state PaymentProcessing {
    [*] --> PendingPayment
    PendingPayment --> RefundIssued : Delivery failed, refund required
    PendingPayment --> PaymentComplete : Payment successful before delivery
    PaymentComplete --> DeliveredSuccessfully
    RefundIssued --> [*]
}

DeliveredSuccessfully --> [*]

state Clarification {
    [*] --> PendingClarification
    PendingClarification --> Dispatched : Delivery scheduled
    PendingClarification --> Cancelled : Order cancelled after 5 days
}

OrderPlacement --> Clarification : Delivery not made within 5 days

state CustomerAction {
    [*] --> OrderActive
    OrderActive --> OrderSuspended : Customer suspends order
    OrderSuspended --> OrderResumed : Customer resumes order
    OrderSuspended --> OrderCancelled : Customer cancels order
    OrderResumed --> ProcessingResumed : Resume order and delivery
    OrderCancelled --> CancelledOrder : Remove specific order, delivery continues
}

OrderPlacement --> CustomerAction : Customer actions impact order
Transit --> CustomerAction : Customer actions during transit
PaymentProcessing --> CustomerAction : Payment-related actions

Clarification --> OrderPlacement : Resolved, return to placement
Clarification --> Transit : Resolved, resume delivery
Clarification --> Cancelled : Unresolved, order cancelled

@enduml
