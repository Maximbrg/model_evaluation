@startuml
title "Cheers System - Enhanced Use-Case Diagram"

left to right direction

actor "Customers" as Customers
actor "Sales Staff" as SalesStaff
actor "Marketing Employees" as MarketingEmp
actor "Gefen System" as Gefen

rectangle "Cheers System" {
    usecase "Place Order" as PlaceOrder
    usecase "Save Customer Details" as SaveCustomerDetails
    usecase "Assign Order to Employee" as AssignOrderToEmployee
    usecase "Urgent Order" as UrgentOrder
    usecase "Regular Order" as RegularOrder
    usecase "Handle Joint Orders" as JointOrders
    usecase "Track Joint Order Details" as TrackJointOrder
    usecase "Browse Available Wines" as BrowseWines
    usecase "Generate Unproductive Employees Report" as UnprodEmployeesReport
    usecase "Generate Current Inventory Report" as CurrentInventoryReport
    usecase "Generate Wine Recommendation Report" as WineRecReport
    usecase "Update Wine Information and types" as UpdateWineInfo
    usecase "Modify Price and Images" as ModifyPrice
    usecase "Recommend Occasions" as RecommendOccasions
    usecase "Link Wine to Type" as LinkWineToType
    usecase "Import XML Data from Gefen" as ImportXMLData
    usecase "Update Inventory and Wine Producers" as UpdateInventory

    PlaceOrder --> SaveCustomerDetails : <<include>>
    PlaceOrder --> AssignOrderToEmployee : <<include>>
    PlaceOrder --> UpdateInventory : <<include>>
    RegularOrder<|-- UrgentOrder : <<extend>>
    PlaceOrder <|-- RegularOrder : <<extend>>
    PlaceOrder <|-- JointOrders : <<extend>>
    JointOrders --> TrackJointOrder : <<include>>
    ImportXMLData --> UpdateInventory : <<include>>

    UnprodEmployeesReport --> SalesStaff : "Generate"
    CurrentInventoryReport --> Gefen : "Send JSON weekly"
    WineRecReport --> SalesStaff : "Generate"

}

note right of UrgentOrder: Tracks priority (1-5) and expected delivery time
note right of JointOrders: Allows multiple customers to combine orders
note right of TrackJointOrder: Tracks wines, quantities, and customers in joint orders
note right of CurrentInventoryReport: Sent to Gefen system in JSON weekly
note right of WineRecReport: Generated by Sales Staff based on customer preferences
note right of UnprodEmployeesReport: Generated by Sales Staff based on orders

Customers <-- WineRecReport : "Get recommendations"

SalesStaff --> PlaceOrder : "Submit orders for customers"
SalesStaff --> BrowseWines : "View wines"
SalesStaff --> PlaceOrder : "Place Employee Order"

MarketingEmp --> UpdateWineInfo : "Update wine information and types"
MarketingEmp --> LinkWineToType : "Link wine to type"
MarketingEmp --> ModifyPrice : "Modify price or images"
MarketingEmp --> RecommendOccasions : "Recommend occasions"
MarketingEmp --> PlaceOrder : "Place Employee Order"

Gefen --> ImportXMLData : "Provides XML data"
@enduml
