' Define Enumerations
enum OrderStatus {
  in_process
  dispatched
  delivered
  paid
  suspended
  canceled
}

enum SweetnessLevel {
  dry
  semi_dry
  semi_sweet
  sweet
}

enum LocationType {
  INDOOR
  OUTDOOR
}

' Person and its subclasses
class Person {
    - personNumber: int
    - fName: string
    - lName: string
    - phone: string
    - address: string
    - email: string
}

class Customer {
    - deliveryAddress: string
    - firstContactDate: date
}

Person <|-- Customer : {complete, overlapping}

class Employee {
    - officeAddress: string
    - employmentStartDate: date
    + viewWineInformation()
}

Person <|-- Employee : {complete, overlapping}

class SalesEmployee {
    + placeOrder()
    + manageCustomerInformation(customer: Customer): void
    + generateUnproductiveEmployeesReport(): Report
    + generateWineRecommendationReportForCustomers(): Report
    + processUrgentOrder(order: UrgentOrder): void
    + processJointOrder(order: JointOrder): void
}

class MarketingEmployee {
    + manageInventory()
    + updateWineInformation(wine: Wines): void
    + updateTypeWineInformation(wineType: WineType): void
    + linkWinesToTypes(wine: Wines, wineType: WineType): void
}

Employee <|-- SalesEmployee : {incomplete, disjoint}
Employee <|-- MarketingEmployee : {incomplete, disjoint}

' Order and its subclasses

    + chooseMainCustomer(): Customer
}
