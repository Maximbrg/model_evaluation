title "Cheers System Class Diagram"

class Order {
  - orderNumber: int
  - orderDate: Date
  - status: OrderStatus
  - shipmentDate: Date
}

class OrderStatus {
  <<enumeration>>
  - IN_PROCESS
  - DISPATCHED
  - DELIVERED
  - PAID
  - SUSPENDED
  - CANCELED
}

class Employee {
  - id: int
  - name: String
  - phoneNumber: String
  - officeAddress: String
  - email: String
  - employmentStartDate: Date
}

class Customer {
  - id: int
  - name: String
  - phoneNumber: String
  - deliveryAddress: String
  - email: String
  - dateOfFirstContact: Date
}

class Wine {
  - id: int
  - name: String
  - serialNumber: String
  - type: WineType
  - pricePerBottle: double
  - sweetnessLevel: SweetnessLevel
  - productImage: String
}

class WineType {
  - id: int
  - name: String
  - description: String
}

class Manufacturer {
  - id: int
  - name: String
  - contactPhoneNumber: String
  - address: String
  - email: String
}

class OrderItem {
  - quantity: int
  - wine: Wine
  - order: Order
}

class OrderAssignment {
  - employee: Employee
  - order: Order
  - assignmentDate: Date
}

class CustomerOrder {
  - customer: Customer
  - order: Order
  - orderDate: Date
}

class WineManufacturer {
  - wine: Wine
  - manufacturer: Manufacturer
  - productionYear: int
}

class StorageLocation {
  - id: int
  - name: String
  - wines: List<Wine>
}

class Inventory {
  - storageLocation: StorageLocation
  - wine: Wine
  - quantity: int
}

class Report {
  <<abstract>>
  - reportType: ReportType
}

class UnproductiveEmployeesReport extends Report {
  - timePeriod: DateRange
  - employees: List<Employee>
}

class CurrentInventoryReportByWine extends Report {
  - wines: List<Wine>
  - inventoryLevels: List<int>
}

class WineRecommendationReportForCustomers extends Report {
  - customer: Customer
  - recommendedWines: List<Wine>
}
